CREATE DATABASE SQL_FINAL_EVALUATION;

USE SQL_FINAL_EVALUATION;

CREATE TABLE STUDENT (
	NUMSTUD INT AUTO_INCREMENT PRIMARY KEY,
	NAME CHAR(255) NOT NULL,
	FIRSTNAME CHAR(255) NOT NULL,
	DATEOFBIRTH DATE,
	STREET CHAR(255),
	POSTALCOD CHAR(255),
	CITY CHAR(255)
);

CREATE TABLE MATTER (
	CODEMAT CHAR(255) PRIMARY KEY,
	WORDING CHAR(255),
	COEF FLOAT
);

CREATE TABLE TEST (
	NUMTEST INT AUTO_INCREMENT PRIMARY KEY,
	TESTDATE DATE,
	PLACE CHAR(255),
	CODEMAT CHAR(255),
	FOREIGN KEY (CODEMAT) REFERENCES MATTER(CODEMAT)
);

CREATE TABLE NOTATION (
	SCORE FLOAT,
	NUMSTUD INT,
	NUMTEST INT,	
	FOREIGN KEY (NUMSTUD) REFERENCES STUDENT(NUMSTUD),
	FOREIGN KEY (NUMTEST) REFERENCES TEST(NUMTEST),
	PRIMARY KEY (NUMSTUD, NUMTEST)
);
